def choose_sorting_algorithm(arr, n):
    """排序算法选择决策"""
    
    # 小数据 (n < 50)
    if n < 50:
        if is_nearly_sorted(arr):
            return "Insertion Sort"  # O(n)
        return "Insertion Sort"
    
    # 整数且范围小
    if is_integer(arr):
        range_size = max(arr) - min(arr) + 1
        if range_size < n * 10:
            return "Counting Sort"  # O(n+k)
        
        digits = len(str(max(arr)))
        if digits <= 6:
            return "Radix Sort"  # O(d(n+k))
    
    # 浮点数均匀分布
    if is_uniform_distribution(arr):
        return "Bucket Sort"  # O(n+k)
    
    # 需要稳定性
    if need_stability:
        return "Merge Sort"  # O(n log n) stable
    
    # 通用场景
    return "Quick Sort"  # O(n log n) 平均最快
